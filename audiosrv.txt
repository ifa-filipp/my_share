def safe_search_mask(series: pl.Series, search_text: str) -> pl.Series:
    if not search_text or not search_text.strip():
        return pl.Series([True] * len(series))
    expr = search_text.strip().lower()
    try:
        groups = {}
        def replace_group(match):
            gid = f"__G{len(groups)}__"
            groups[gid] = match.group(1).lower()
            return gid
        expr_proc = re.sub(r'\(([^)]+)\)', replace_group, expr)
        expr_proc = expr_proc.replace(' и ', ' & ').replace(' или ', ' | ')
        for gid, subexpr in groups.items():
            subexpr = subexpr.replace(' и ', ' & ').replace(' или ', ' | ')
            expr_proc = expr_proc.replace(gid, f"({subexpr})")
        atom_pattern = r"[^&|()]+"
        atoms = [a.strip() for a in re.findall(atom_pattern, expr_proc) if a.strip()]
        atoms = list(dict.fromkeys(atoms))
        local_vars = {}
        for i, atom in enumerate(atoms):
            var_name = f"term_{i}"
            phrase = atom.replace('&', '').replace('|', '').strip()
            if not phrase:
                local_vars[var_name] = pl.Series([True] * len(series))
            else:
                # Use (?i) flag for case-insensitive matching in Polars regex
                mask = series.str.contains(f"(?i){re.escape(phrase)}", literal=False).fill_null(False)
                local_vars[var_name] = mask
            expr_proc = expr_proc.replace(atom, var_name)
        local_vars['&'] = lambda a, b: a & b
        local_vars['|'] = lambda a, b: a | b
        mask = eval(expr_proc, {"__builtins__": {}}, local_vars)
        return mask if isinstance(mask, pl.Series) else pl.Series([bool(mask)] * len(series))
    except Exception:
        words = expr.split()
        if not words:
            return pl.Series([True] * len(series))
        mask = pl.Series([True] * len(series))
        for word in words:
            w = word.strip()
            if w:
                # Use (?i) flag for case-insensitive matching in Polars regex
                mask &= series.str.contains(f"(?i){re.escape(w)}", literal=False).fill_null(False)
        return mask

def filter_dialog_by_type_codes(dialog: str, type_codes: list[int]) -> str:
    """
    Извлекает из диалога только сообщения указанных типов.
    
    Формат диалога:
        "K - (1.5s – 4.6s) - текст сообщения...\nC - (5.0s – 17.7s) - текст..."
    
    Маппинг кодов → букв в диалоге:
        0 → 'K' (клиент)
        1 → 'C' (сотрудник)
    
    Возвращает строку с отфильтрованными сообщениями (или пустую строку).
    """
    if not dialog or not isinstance(dialog, str):
        return ""
    
    # Маппинг: код типа → буква в начале сообщения
    code_to_letter = {0: 'K', 1: 'C'}
    target_letters = {code_to_letter[c] for c in type_codes if c in code_to_letter}
    
    if not target_letters:
        return ""
    
    # 🔧 ИСПРАВЛЕНИЕ: используем findall вместо split для надёжного парсинга
    # Паттерн захватывает ВСЁ сообщение целиком: буква + время + текст до следующего сообщения или конца
    pattern = r'([CK])\s*-\s*\([^)]+\)\s*-\s*[^K\n]*(?=(?:\n?[CK]\s*-\s*\(|$))'
    
    matches = re.findall(pattern, dialog, re.DOTALL)
    
    # re.findall с группами возвращает кортежи, если есть группы захвата
    # Поэтому используем другой подход: ищем полные блоки
    pattern_full = r'([CK])\s*-\s*\([^)]+\)\s*-\s*(.*?)(?=\n?[CK]\s*-\s*\(|$)'
    
    filtered_parts = []
    for match in re.finditer(pattern_full, dialog, re.DOTALL):
        speaker = match.group(1)  # 'K' или 'C'
        text = match.group(2).strip()  # текст сообщения
        full_msg = match.group(0).strip()  # полное сообщение
        
        if speaker in target_letters:
            filtered_parts.append(full_msg)
    
    return '\n'.join(filtered_parts) if filtered_parts else ""

def get_trigrams(s: str):
    s = " " + s.strip().lower() + " "  # граничные маркеры улучшают качество
    return {s[i:i+3] for i in range(len(s) - 2)}

def jaccard_similarity(set1, set2):
    if not set1 and not set2:
        return 1.0
    intersection = len(set1 & set2)
    union = len(set1 | set2)
    return intersection / union

def fuzzy_trigram_match(text_series: pl.Series, query_word: str, threshold=0.8) -> pl.Series:
    """
    Performs fuzzy matching using trigram Jaccard similarity against    Returns a boolean Series indicating matches.
    """
    query_trigrams = get_trigrams(query_word)

    def match_cell(cell_value):
        if cell_value is None:
            return False
        # Split the string into words and check each one
        words = str(cell_value).split()
        for w in words:
            if jaccard_similarity(query_trigrams, get_trigrams(w)) >= threshold:
                return True
        return False

    # Use map_elements to apply the custom function to each element in the Series
    return text_series.map_elements(match_cell, return_dtype=pl.Boolean)

def apply_sql_dialog_filter(df: pl.DataFrame, sql_where: str) -> pl.DataFrame:
    """Применяет SQL-условие к Polars DataFrame через DuckDB"""
    if not sql_where or not sql_where.strip():
        return df
    try:
        con = duckdb.connect()
        # Регистрируем датафрейм под именем 'transcripts' для SQL
        con.register("transcripts", df)
        
        # Формируем запрос. Если пользователь ввёл только условие WHERE, оборачиваем его
        sql_clean = sql_where.strip()
        if sql_clean.upper().startswith("SELECT"):
            query = sql_clean
        else:
            query = f"SELECT * FROM transcripts WHERE {sql_clean}"
            
        # Выполняем запрос и возвращаем результат обратно в Polars
        result_df = con.execute(query).pl()
        return result_df
    except Exception as e:
        st.error(f"⚠️ Ошибка в SQL-условии: {str(e)}")
        # Возвращаем пустой DF, чтобы таблица не показывала некорректные данные
        return df.filter(pl.lit(False))
        
def apply_filters_logic(full_df: pl.DataFrame, filters_dict: dict):
    """Применяет фильтры и возвращает отфильтрованный датафрейм и список чатов"""

    # Копируем датафрейм (Polars не мутирует оригинальный DataFrame, так что копия не нужна)
    filtered_df = full_df.clone()

    # Фильтр по дате
    s, e = filters_dict['date_actual']
    if not isinstance(s, date):
        s = s.date()
    if not isinstance(e, date):
        e = e.date()
    filtered_df = filtered_df.filter(
        (pl.col('ddate').dt.date() >= s) &
        (pl.col('ddate').dt.date() <= e)
    )

    # Фильтр по id
    if filters_dict.get('audio'):
        audio_values = [str(x) for x in filters_dict['audio']]
        filtered_df = filtered_df.filter(pl.col('id').cast(pl.Utf8).is_in(audio_values))
        
    # Фильтр по phone
    if filters_dict.get('phone'):
        phone_values = [str(x) for x in filters_dict['phone']]
        filtered_df = filtered_df.filter(pl.col('phone').cast(pl.Utf8).is_in(phone_values))

    # Фильтр по inn (client)
    if filters_dict.get('client'):
        filtered_df = filtered_df.filter(pl.col('inn').is_in(filters_dict['client']))

    # Фильтр по inn (direction)
    if filters_dict.get('direction'):
        filtered_df = filtered_df.filter(pl.col('direction').is_in(filters_dict['direction']))

    # Фильтр по employeename (manager)
    if filters_dict.get('manager'):
        # Создаем список условий: содержит ли столбец каждое из выбранных имен
        # literal=True важен, чтобы точки в ФИО не воспринимались как спецсимволы regex
        conditions = [
            pl.col('employeename').str.contains(name, literal=True) 
            for name in filters_dict['manager']
        ]
        
        # Объединяем условия через ИЛИ (если выбрано несколько менеджеров, 
        # покажет строки, где есть хоть кто-то из списка)
        filtered_df = filtered_df.filter(pl.any_horizontal(conditions))

    # Фильтр по contact_id
    if filters_dict.get('contact_id'):
        contact_id_values = [str(x) for x in filters_dict['contact_id']]
        filtered_df = filtered_df.filter(pl.col('contact_id').cast(pl.Utf8).is_in(contact_id_values))

    # # Фильтр по emotion
    # if filters_dict.get('emotion'):
    #     filtered_df = filtered_df.filter(pl.col('has_emotion').cast(pl.Utf8).is_in(filters_dict['emotion']))
    

    # Доп фильтры
    chat_ids_to_keep = set(filtered_df.select(pl.col('id')).to_series().to_list())
    
    if "Диалог с эмоциональной окраской со стороны клиента" in filters_dict.get('extra', []):
        emotional_ids = set(
            filtered_df
            .filter(pl.col('has_emotion') == 1)
            .select('id')
            .unique()
            .to_series()
            .to_list()
        )
        chat_ids_to_keep &= emotional_ids  # Пересечение множеств
    if "Тематика обращений 1.1" in filters_dict.get('extra', []):
        theme_1_1s = filters_dict.get('theme_1_1', [])
        theme_1_1_ids = set(
            filtered_df
            .filter(pl.col('theme_1_1').is_in(theme_1_1s))
            .select('id') 
            .unique()
            .to_series()
            .to_list()
        )
        chat_ids_to_keep &= theme_1_1_ids  # Пересечение множеств
    if "Тематика обращений 1.2" in filters_dict.get('extra', []):
        theme_1_2s = filters_dict.get('theme_1_2', [])
        theme_1_2_ids = set(
            filtered_df
            .filter(pl.col('theme_1_2').is_in(theme_1_2s))
            .select('id')
            .unique()
            .to_series()
            .to_list()
        )
        chat_ids_to_keep &= theme_1_2_ids  # Пересечение множеств

    if "Вежливость со стороны сотрудника" in filters_dict.get('extra', []):
        polite_ids = set(
            filtered_df
            .filter(pl.col('employee_politeness') == True)
            .select('id')
            .unique()
            .to_series()
            .to_list()
        ) 
        
        include = filters_dict.get('polite_employee_include', True)
        # 3. Фильтрация множеств
        if include:
            chat_ids_to_keep &= polite_ids
        else:
            chat_ids_to_keep -= polite_ids

    # === Фильтр "Обращение по имени" (4 независимых условия) ===
    if "Обращение по имени" in filters_dict.get('extra', []):
        name_asked = filters_dict.get('name_asked', 'Любой')
        name_provided = filters_dict.get('name_provided', 'Любой')
        name_used = filters_dict.get('name_used', 'Любой')
        name_used_3plus = filters_dict.get('name_used_3plus', 'Любой')
        required_cols = [
            'is_name_asked',
            'is_name_provided',
            'is_name_used',
            'is_name_used_3plus'
        ]
        if all(col in filtered_df.columns for col in required_cols):
            def get_chat_ids_bool(df: pl.DataFrame, col: str, target_val: bool) -> set:
                return set(
                    df
                    .filter(
                        pl.col(col)
                        .fill_null(False)
                        .cast(pl.Boolean)
                        == target_val
                    )
                    .select('id')
                    .unique()
                    .to_series()
                    .to_list()
                )
            # 1) Сотрудник спрашивал имя?
            if name_asked == "Да":
                chat_ids_to_keep &= get_chat_ids_bool(filtered_df, 'is_name_asked', True)
            elif name_asked == "Нет":
                chat_ids_to_keep &= get_chat_ids_bool(filtered_df, 'is_name_asked', False)
            # 2) Имя озвучено клиентом?
            if name_provided == "Да":
                chat_ids_to_keep &= get_chat_ids_bool(filtered_df, 'is_name_provided', True)
            elif name_provided == "Нет":
                chat_ids_to_keep &= get_chat_ids_bool(filtered_df, 'is_name_provided', False)
            # 3) Сотрудник обратился по имени?
            if name_used == "Да":
                chat_ids_to_keep &= get_chat_ids_bool(filtered_df, 'is_name_used', True)
            elif name_used == "Нет":
                chat_ids_to_keep &= get_chat_ids_bool(filtered_df, 'is_name_used', False)
            # 4) Сотрудник обратился по имени 3+ раз?
            if name_used_3plus == "Да":
                chat_ids_to_keep &= get_chat_ids_bool(filtered_df, 'is_name_used_3plus', True)
            elif name_used_3plus == "Нет":
                chat_ids_to_keep &= get_chat_ids_bool(filtered_df, 'is_name_used_3plus', False)
        else:
            missing = [c for c in required_cols if c not in filtered_df.columns]
            st.warning(
                f"Не найдены колонки для фильтра 'Обращение по имени': {missing}. Обновите ETL."
            )

    

    # Фильтрация по итоговому списку
    # if chat_ids_to_keep:  # Защита от пустого множества (is_in([]) вернёт пустой DF)
    filtered_df = filtered_df.filter(pl.col('id').is_in(chat_ids_to_keep))

    # === 🔍 ПОИСК ===
    search_words = filters_dict.get('search', '')
    if search_words.strip():
        
        # 🔹 Режим 1: нечёткий поиск по ВСЕМ сообщениям (как у вас было)
        if not filters_dict.get('search_mode_toggle', False):
            words = [w.strip() for w in search_words.split() if w]
            combined_mask = pl.Series([False] * len(filtered_df))
            for word in words:
                word_mask = fuzzy_trigram_match(
                    filtered_df['dialog'].fill_null('').cast(pl.String), 
                    word, 
                    threshold=0.3
                )
                combined_mask |= word_mask
            chat_ids_with_match = filtered_df.filter(combined_mask)['id'].unique()
            filtered_df = filtered_df.filter(pl.col('id').is_in(chat_ids_with_match))
        
        # 🔹 Режим 2: точный поиск ТОЛЬКО в выбранных типах сообщений
        else:
            selected_search_type_codes = filters_dict.get('search_type_codes', [])
            selected_search_type_codes = [int(c) if isinstance(c, str) else c for c in selected_search_type_codes]
            if selected_search_type_codes:
                # 1. Создаём временную колонку: диалог, отфильтрованный по выбранным типам
                filtered_dialogs = (
                    filtered_df['dialog']
                    .fill_null('')
                    .cast(pl.String)
                    .map_elements(
                        lambda d: filter_dialog_by_type_codes(d, selected_search_type_codes),
                        return_dtype=pl.String
                    )
                )
                
                # 2. Применяем ваш safe_search_mask к отфильтрованным диалогам
                mask = safe_search_mask(filtered_dialogs, search_words)
                
                # 3. Находим чаты, где есть совпадения
                matching_chats = filtered_df.filter(mask)['id'].unique()
                
                # 4. Фильтруем исходный датафрейм по этим чатам
                filtered_df = filtered_df.filter(pl.col('id').is_in(matching_chats))
            else:
                # Если типы не выбраны — возвращаем пустой результат
                filtered_df = filtered_df.slice(0, 0)

    def load_file(uploaded_file, original_filename, temp_dir='audio_files'):
        if uploaded_file is None:
            return None, None
    
        temp_dir_full = temp_dir + '/' + original_filename[:-4]
        
        # Создаём папку, если её нет
        os.makedirs(temp_dir_full, exist_ok=True)
    
        # Полный путь к файлу: temp_dir_full/original_filename
        file_path = os.path.join(temp_dir_full, original_filename)
    
    
        # Записываем содержимое
        with open(file_path, "wb") as f:
            f.write(uploaded_file.getvalue())
    
        return str(temp_dir_full), str(original_filename)

    def to_s3(file_path: str):
        """
        Args:

        s3 = S3Bucket.load("s3")
    
        # 6. Загружаем в S3
        s3.upload_from_path(file_path, object_key)
        return object_key

    def send_message(subject_text, message_text, recipients, attachment_path=None):
        # отправка письма на почту
            # Настройки почты
        smtp_server = 'smtp.uralsib.ru'
        smtp_port = 25
        sender_email = 'dmas@uralsib.ru'
                #recipient_email = ['VerbovaMA@uralsib.ru']
                # Создаем объект сообщения
        message = MIMEMultipart()
        message['From'] = sender_email
        message['To'] = recipients
        message['Subject'] = subject_text
            
        body = message_text #(message_text, 'html', 'utf-8'))
        message.attach(MIMEText(body, 'html'))
    
        # Если указан путь к файлу — добавляем вложение

            with open(attachment_path, "rb") as attachment:
                part = MIMEBase('application', 'octet-stream')
                part.set_payload(attachment.read())
            encoders.encode_base64(part)
            part.add_header(
                'Content-Disposition',
                f'attachment; filename="{os.path.basename(attachment_path)}"'
            )
            message.attach(part)
        
        try:
                    # Устанавливаем соединение с сервером
            with smtplib.SMTP(smtp_server, smtp_port) as server:
                server.sendmail(sender_email, recipients, message.as_string())
                server.quit()
                print("Письмо успешно отправлено!")
